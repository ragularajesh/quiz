# app.py
from flask import Flask, jsonify, request, send_from_directory
import json
import random
import os

app = Flask(__name__, static_folder="static", static_url_path="/static")

QUESTIONS_FILE = "questions.json"

def load_questions():
    if not os.path.exists(QUESTIONS_FILE):
        return []
    with open(QUESTIONS_FILE, "r", encoding="utf-8") as f:
        return json.load(f)

def save_questions(qs):
    with open(QUESTIONS_FILE, "w", encoding="utf-8") as f:
        json.dump(qs, f, indent=2, ensure_ascii=False)

@app.route("/")
def index():
    return send_from_directory("static", "index.html")

@app.route("/api/topics")
def topics():
    qs = load_questions()
    topics = sorted({q["topic"] for q in qs})
    difficulties = sorted({q["difficulty"] for q in qs})
    return jsonify({"topics": topics, "difficulties": difficulties})

@app.route("/api/generate", methods=["POST"])
def generate():
    data = request.get_json() or {}
    topic = data.get("topic")
    difficulty = data.get("difficulty")
    n = int(data.get("count", 5))
    qs = load_questions()

    # filter
    filtered = [q for q in qs if (topic in (None, "", "Any") or q["topic"] == topic)
                         and (difficulty in (None, "", "Any") or q["difficulty"] == difficulty)]
    if len(filtered) == 0:
        return jsonify({"error": "No questions match the chosen filters."}), 400

    n = min(n, len(filtered))
    selected = random.sample(filtered, n)

    # remove answers before sending
    sent = []
    for q in selected:
        qcopy = {k: q[k] for k in ("id", "text", "choices", "topic", "difficulty")}
        sent.append(qcopy)

    return jsonify({"questions": sent})

@app.route("/api/answer-check", methods=["POST"])
def answer_check():
    payload = request.get_json() or {}
    answers = payload.get("answers", {})  # {id: selected_index}
    qs = load_questions()
    qmap = {q["id"]: q for q in qs}
    results = []
    score = 0
    for qid_str, sel in answers.items():
        try:
            qid = int(qid_str)
        except:
            qid = qid_str
        q = qmap.get(qid)
        if not q:
            continue
        correct = q["answer_index"]
        is_correct = (int(sel) == int(correct))
        if is_correct:
            score += 1
        results.append({
            "id": qid,
            "selected": int(sel),
            "correct": int(correct),
            "is_correct": is_correct,
            "text": q["text"],
            "choices": q["choices"]
        })
    return jsonify({"score": score, "total": len(results), "results": results})

@app.route("/api/add-question", methods=["POST"])
def add_question():
    payload = request.get_json() or {}
    required = ["text", "choices", "answer_index", "topic", "difficulty"]
    if not all(k in payload for k in required):
        return jsonify({"error": f"Required fields: {required}"}), 400

    qs = load_questions()
    new_id = max([q["id"] for q in qs], default=0) + 1
    newq = {
        "id": new_id,
        "text": payload["text"],
        "choices": payload["choices"],
        "answer_index": int(payload["answer_index"]),
        "topic": payload["topic"],
        "difficulty": payload["difficulty"]
    }
    qs.append(newq)
    save_questions(qs)
    return jsonify({"status": "ok", "id": new_id})

if __name__ == "__main__":
    # create sample file if missing

            {"id":5, "text":"What does HTTP stand for?","choices":["HyperText Transfer Protocol","HighText Transfer Protocol","HyperText Transmission Protocol","Hyper Transfer Text Prot"], "answer_index":0, "topic":"Computing","difficulty":"Easy"}
        ]
        save_questions(sample)
    app.run(debug=True)

async function init() {
  const topicSel = document.getElementById("topic");
  const diffSel = document.getElementById("difficulty");

  const r = await fetch("/api/topics");
  const data = await r.json();
  addOptions(topicSel, ["Any", ...data.topics]);

}

function addOptions(selectEl, arr) {
  selectEl.innerHTML = "";
  arr.forEach(a => {
    const o = document.createElement("option");
    o.value = a;
    o.textContent = a;

}

      label.appendChild(input);
      const span = document.createElement("span");
      span.textContent = c;
      label.appendChild(span);
      wrapper.appendChild(label);
    });
    form.appendChild(wrapper);
  });
}

How to run locally

  1. Create a project folder, put app.py and questions.json (optional) at root and a folder named static containing index.html and styles.css.

  2. Create a Python venv (recommended) and install Flask:

    bash
    python -m venv venv
source venv/bin/activate   # or venv\Scripts\activate on Windows
pip install flask

  3. Run:

    nginx
    python app.py

  4. Open http://127.0.0.1:5000 in your browser.

@app.route("/api/topics")
def topics():
    topics = sorted({q["topic"] for q in QUESTIONS})
    diffs = sorted({q["difficulty"] for q in QUESTIONS})
    return jsonify({"topics":topics,"difficulties":diffs})

@app.route("/api/generate", methods=["POST"])
def generate():
    data = request.get_json()
    topic = data["topic"]
    difficulty = data["difficulty"]
    count = int(data["count"])

    filtered = [q for q in QUESTIONS
                if (topic=="Any" or q["topic"]==topic)
                and (difficulty=="Any" or q["difficulty"]==difficulty)]

    if len(filtered)==0:
        return jsonify({"error":"No questions match filters"}), 400

    selected = random.sample(filtered, min(count,len(filtered)))
    send = [{"id":q["id"],"text":q["text"],"choices":q["choices"]} for q in selected]

    return jsonify({"questions":send})

@app.route("/api/check", methods=["POST"])
def check():
    data = request.get_json()
    answers = data["answers"]

    score=0
    results=[]

    qmap={q["id"]:q for q in QUESTIONS}

    for qid, sel in answers.items():
        q=qmap[int(qid)]
        sel=int(sel)
        correct=q["answer_index"]
        ok=(sel==correct)
        if ok: score+=1

        results.append({
            "text":q["text"],
            "choices":q["choices"],
            "selected":sel,
            "correct":correct,
            "is_correct":ok
        })

    return jsonify({"score":score,"total":len(results),"results":results})
